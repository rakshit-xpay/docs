title: "Active"

---

  {" "}

  tracked and processed individually.{" "}

  {" "}

  {" "}

  {" "}

  {" "}

  in this event.{" "}

  {" "}

</ResponseField> <ResponseField name="currency" type="string">
  {" "}

  {" "}
